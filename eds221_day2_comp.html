Matrices

class(fish_size)

[1] "double"

fish_size[1,1]

what happens if we create a matrix with different type of data

#converts all to character 
fish_type2 <- matrix(c(12, "trout", 18, 13L), ncol =2, nrow =2, byrow =FALSE)
class(fish_type2)
[1] "matrix" "array" 
typeof(fish_type2)
[1] "character"
#look at second row but keep all columns
fish_type2[2,]

fish_type2[,1]
